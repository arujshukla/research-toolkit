\begin{frame}{A Standard Content Slide}
  \begin{columns}[T,onlytextwidth]
    \column{0.58\textwidth}
      \begin{itemize}\tightlist
        \item Use \highlight{short declarative bullets} rather than full paragraphs.
        \item Reserve bold color for the main idea on the slide.
        \item Keep one claim or chart per frame when possible.
      \end{itemize}

    \column{0.38\textwidth}
      \begin{block}{Rule of Thumb}
        If a slide needs narration to be understood, it is probably doing too much.

  \end{columns}
\end{frame}
